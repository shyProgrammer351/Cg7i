 프로그램 종료 
 
 가위바위보 선택 함수 
    사용자가 가위바위보중 하나를 선택 
    가위바위보 선택 결과를 출력 
 
 A양이 승리한 경우 확인 함수 
    A양이 가위이고 B군이 보를 낸 경우 
    A양이 바위이고 B군이 가위를 낸 경우 
    A양이 보이고 B군이 바위를 낸 경우

	// 파일명 : ./Chapter17/RockPaperScissors.c

	#include <stdio.h>

		 

	// 논리값 매크로 상수 정의

	#define TRUE   1

	#define FALSE  0

	#define SCISSOR 1

	#define ROCK    2

	#define PAPER   3

		 

	// A양이 승리한 경우 확인 함수 

F1	#define DoesPlayerAWin( playerA, playerB )

		// A양이 가위이고 B군이 보를 낸 경우 

		( ( ( playerA == SCISSOR ) && ( playerB == PAPER ) )

			// A양이 바위이고 B군이 가위를 낸 경우 

			|| ( ( playerA == ROCK ) && ( playerB == SCISSOR ) )

			// A양이 보이고 B군이 바위를 낸 경우 

			|| ( ( playerA == PAPER ) && ( playerB == ROCK ) ) )

		? TRUE : FALSE

	 

	// 가위바위보 선택 함수 

F2b	int getRockPaperScissors() {

		int player = 0;

		printf( "가위(1), 바위(2), 보(3) 중 하나를 숫자로 입력하세요: " );

F21		scanf( "%d", &player );

	 

		// 가위바위보 선택 결과를 출력 

F22		switch( player ) {

F23		case SCISSOR:

			printf( "가위를 냈습니다.\n" );

F24			break;

F25		case ROCK:

			printf( "바위를 냈습니다.\n" );

F26			break;

F27		case PAPER:

			printf( "보를 냈습니다.\n" );

		}

			 

F28		return player;

F2e	}
