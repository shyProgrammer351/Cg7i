	printf( "가위(1), 바위(2), 보(3) 중 하나를 숫자로 입력하세요: " ); 
	int player = 0;	 
	scanf( "%d", &player ); 	 
 
	( player ) { 
	 SCISSOR:									 
		printf( "가위를 냈습니다.\n" );  
		; 
	 ROCK:

		; 
	 PAPER:									 
		printf( "보를 냈습니다.\n" );  
	} 
		 
	 player; 
}
