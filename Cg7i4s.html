문제

공을 3개 컵 중에서 어디에 숨겼는지 사용자가 맞추는 문제입니다 실행순서를 클릭하세요 
  ___    ___    ___  
 |   |  |   |  |   | 
 | 1 |  | 2 |  | 3 | 
   O 
 찾았다!
 1, 2, 3 중에서 공을 숨긴 컵을 찾으세요 : 1

프로그램 코드

	#include <stdio.h>

	#include <stdlib.h>

	#include <time.h>


	#define DEBUG


1	int main() {

		int ball = 0, cup = 0;

		

		srand( time( NULL ) );

2		ball = rand() % 3 + 1;

3		#ifdef DEBUG

4		printf( " \033[8;1f [DEBUG] 공 위치 %d \033[1;1f", ball);

5		#endif

		

		printf( "  ___    ___    ___  \n" );

		printf( " |   |  |   |  |   | \n" );

		printf( " | 1 |  | 2 |  | 3 | \n\n\n\n" );


		printf( " 1, 2, 3 중에서 공을 숨긴 컵을 찾으세요 : " );

6		scanf( "%d", &cup );

			

		printf( " \033[4;%dfO\n \033[5;%df", ball*7-3, cup*7-5 );

7		( cup == ball ) ? printf( "찾았다!\n") : printf( "놓쳤다!\n" );


8		return 0;

	}
